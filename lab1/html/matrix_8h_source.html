126 int i1 = getVertexIndex(v1);

128 if (i1 == -1 || i2 == -1) return -1;

130 const double INF = 1e9;
131 vector<double> dist(vertices.size(), INF);
132 vector<bool> visited(vertices.size(), false);
133 priority_queue<pair<double,int>, vector<pair<double,int>>, greater<>> pq;
134
135 dist[i1] = 0;
136 pq.push({0, i1});
137
138 while (!pq.empty()) {
139 int u = pq.top().second; pq.pop();
140 if (u < 0 || u >= vertices.size()) continue;
141 if (visited[u]) continue;
142 visited[u] = true;
143
144 for (int v = 0; v < vertices.size(); v++) {
145 if (adjMatrix[u][v] != 0 && dist[u] + adjMatrix[u][v] < dist[v]) {
146 dist[v] = dist[u] + adjMatrix[u][v];
147 pq.push({dist[v], v});
148 }
149 }
150 }
151
152 return dist[i2] == INF ? -1 : dist[i2];
