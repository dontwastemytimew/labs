174 int i1 = getVertexIndex(v1);

176 if (i1 == -1 || i2 == -1) return -1;

178 const double INF = 1e9;
179 vector<double> dist(vertices.size(), INF);
180 vector<bool> visited(vertices.size(), false);
181 priority_queue<pair<double,int>, vector<pair<double,int>>, greater<>> pq;
182
183 dist[i1] = 0;
184 pq.push({0, i1});
185
186 while (!pq.empty()) {
187 int u = pq.top().second; pq.pop();
188 if (u < 0 || u >= vertices.size()) continue;
189 if (visited[u]) continue;
190 visited[u] = true;
191
192 for (auto& [v, w] : adjList[u]) {
193 if (v < 0 || v >= vertices.size()) continue;
194 if (dist[u] + w < dist[v]) {
195 dist[v] = dist[u] + w;
196 pq.push({dist[v], v});
197 }
198 }
199 }
200
201 return dist[i2] == INF ? -1 : dist[i2];
