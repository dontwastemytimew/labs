184 vector<VertexT> computeOptimalRoute(int vehicleIndex, const VertexT& from, const VertexT& to) const {
185 vector<VertexT> route;
186
187 if (vehicleIndex < 0 || vehicleIndex >= vehicles.size()) return route;
188
189 int indexFrom = graph.getVertexIndex(from);
190 int indexTo = graph.getVertexIndex(to);
191
192 if (indexFrom == -1 || indexTo == -1) return route;
193
194 vector<double> dist(graph.graphSize(), 1e9);
195 vector<int> prev(graph.graphSize(), -1);
196
197 dist[indexFrom] = 0;
198 priority_queue<pair<double,int>, vector<pair<double,int>>, greater<>> pq;
199 pq.push({0, indexFrom});
200
201 while (!pq.empty()) {
202 int u = pq.top().second; pq.pop();
203 for (auto& [v, w] : graph.getAdj(u)) {
204 if (dist[u] + w < dist[v]) {
205 dist[v] = dist[u] + w;
206 prev[v] = u;
207 pq.push({dist[v], v});
208 }
209 }
210 }
211
212 int cur = indexTo;
213 if (prev[cur] == -1 && indexFrom != indexTo) return route;
214
215 while (cur != -1) {
216 route.insert(route.begin(), graph.getVertex(cur));
217 cur = prev[cur];
218 }
219
220 return route;
221 }
